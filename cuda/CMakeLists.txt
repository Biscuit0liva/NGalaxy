    galaxy_kernel.cu
)

    ${CMAKE_SOURCE_DIR}/common
    ${CMAKE_SOURCE_DIR}/glad/include
    ${CMAKE_SOURCE_DIR}/glfw/include
)

target_link_libraries(galaxy_cuda PRIVATE
    common
    glfw
    glad
    OpenGL::GL
)
